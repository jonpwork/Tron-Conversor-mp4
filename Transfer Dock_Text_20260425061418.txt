<script>
const state = { img:null, aud:null, res:'720x1280', leg:'nenhuma', music:false };
let audioObj = null;

function toggleMusic() {
  const btn = document.getElementById('somBtn');
  if(!audioObj) {
    audioObj = new Audio('https://cdn.pixabay.com/download/audio/2022/10/18/audio_31c2730ebb.mp3');
    audioObj.loop = true; audioObj.volume = 0.4;
  }
  if(state.music) { audioObj.pause(); btn.classList.remove('on'); }
  else { audioObj.play(); btn.classList.add('on'); }
  state.music = !state.music;
}

function upFile(el, type) {
  const f = el.files[0]; if(!f) return;
  state[type] = f;
  document.getElementById('name'+(type==='img'?'Img':'Aud')).textContent = f.name.slice(0,10)+'...';
}

document.querySelectorAll('.opt').forEach(o => {
  o.onclick = () => {
    document.querySelectorAll('.opt[data-group="'+o.dataset.group+'"]').forEach(x=>x.classList.remove('sel'));
    o.classList.add('sel');
    state[o.dataset.group] = o.dataset.val;
  }
});

async function gerarVideo() {
  if(!state.img || !state.aud) return alert('Selecione os arquivos!');
  const lbl = document.getElementById('centerLabel');
  const pct = document.getElementById('centerPct');
  const fill = document.getElementById('progressFill');
  document.getElementById('progressBar').style.display='block';
  lbl.style.display='none'; pct.style.display='block';
  
  let p = 0;
  const iv = setInterval(() => {
    p = Math.min(p + 3, 98);
    pct.textContent = Math.round(p)+'%';
    fill.style.width = p+'%';
  }, 150);

  setTimeout(() => {
    clearInterval(iv);
    pct.textContent = '100%'; fill.style.width = '100%';
    alert('Vídeo Gerado!');
    lbl.style.display='block'; pct.style.display='none';
  }, 5000);
}

// Visuals
const cv = document.getElementById('particles');
const ctx = cv.getContext('2d');
let W, H;
function rs(){ W=cv.width=window.innerWidth; H=cv.height=window.innerHeight; }
window.onresize=rs; rs();
const pts = [];
for(let i=0; i<30; i++) pts.push({x:Math.random()*W, y:Math.random()*H, r:Math.random()*1.5});
function draw(){
  ctx.clearRect(0,0,W,H); ctx.fillStyle='rgba(0,212,255,0.2)';
  pts.forEach(p => { ctx.beginPath(); ctx.arc(p.x, p.y, p.r, 0, 7); ctx.fill(); p.y-=0.3; if(p.y<0) p.y=H; });
  requestAnimationFrame(draw);
}
draw();
